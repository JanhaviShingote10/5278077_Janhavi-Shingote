#include <stdio.h>

#define MAX_VAL 100

void countingSort(int arr[], int n, int freq[]) {
    // Initialize frequency array
    for (int i = 0; i < MAX_VAL; i++) {
        freq[i] = 0;
    }

    // Count occurrences
    for (int i = 0; i < n; i++) {
        freq[arr[i]]++;
    }
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int freq[MAX_VAL];
    countingSort(arr, n, freq);

    // Output frequency array
    for (int i = 0; i < MAX_VAL; i++) {
        printf("%d", freq[i]);
        if (i != MAX_VAL - 1) printf(" ");
    }
    printf("\n");

    return 0;
}
