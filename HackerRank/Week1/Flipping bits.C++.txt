#include <stdio.h>
#include <stdint.h>  // for uint32_t

unsigned int flipBits(unsigned int num) {
    return num ^ 0xFFFFFFFF;  // XOR with all 1's flips the bits
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        unsigned int value;
        scanf("%u", &value);
        printf("%u\n", flipBits(value));
    }

    return 0;
}
